/**Se crea la tabla del Zona**/

CREATE TABLE Zona(
Nombre_Zona VARCHAR (20) NOT NULL,
Nombre_Concesionario VARCHAR (20) NOT NULL,
PRIMARY KEY (Nombre_Zona,Nombre_Concesionario),
CONSTRAINT fk_zn_co FOREIGN KEY(Nombre_Concesionario) REFERENCES Concesionario (Nombre_Concesionario)
);

/**Se crea la tabla del micro bus**/

CREATE TABLE Microbus(
Chasis VARCHAR(20) NOT NULL,
Nombre_Concesionario VARCHAR(255) NOT NULL,
Codigo VARCHAR(8) NOT NULL,
Marca VARCHAR(20) NOT NULL,
Placa VARCHAR(6) NOT NULL,
Modelo INT NOT NULL,
Linea VARCHAR(20) NOT NULL,
Accesible VARCHAR(10) NOT NULL,
Fecha_de_Vinculacion DATE NOT NULL,
Fecha_de_Matricula DATE NOT NULL,
Nivel_de_Emision VARCHAR(20) NOT NULL,
GPS VARCHAR(10) NOT NULL,
Fecha_de_Varada DATE NULL,
PRIMARY KEY (Chasis,Nombre_Concesionario),
CONSTRAINT fk_mb_co FOREIGN KEY(Nombre_Concesionario) REFERENCES Concesionario (Nombre_Concesionario)
ON UPDATE NO ACTION ON DELETE CASCADE
);

SELECT *FROM Microbus;
/**Se crea la tabla del Bus**/

CREATE TABLE Bus(
Chasis VARCHAR(20) NOT NULL,
Nombre_Concesionario VARCHAR(255) NOT NULL,
Codigo VARCHAR(8) NOT NULL,
Marca VARCHAR(20) NOT NULL,
Placa VARCHAR(6) NOT NULL,
Modelo INT NOT NULL,
Linea VARCHAR(20) NOT NULL,
Accesible VARCHAR(10) NOT NULL,
Fecha_de_Vinculacion DATE NOT NULL,
Fecha_de_Matricula DATE NOT NULL,
Nivel_de_Emision VARCHAR(20) NOT NULL,
AC VARCHAR(10) NOT NULL,
Fecha_de_Varada DATE NULL,
PRIMARY KEY (Chasis,Nombre_Concesionario),
CONSTRAINT fk_bu_co FOREIGN KEY(Nombre_Concesionario) REFERENCES Concesionario (Nombre_Concesionario)
ON UPDATE NO ACTION ON DELETE CASCADE
);

/**Se crea la tabla de Buseta**/

CREATE TABLE Buseta(
Chasis VARCHAR(20) NOT NULL,
Nombre_Concesionario VARCHAR(255) NOT NULL,
Codigo VARCHAR(8) NOT NULL,
Marca VARCHAR(20) NOT NULL,
Placa VARCHAR(6) NOT NULL,
Modelo INT NOT NULL,
Linea VARCHAR(20) NOT NULL,
Accesible VARCHAR(10) NOT NULL,
Fecha_de_Vinculacion DATE NOT NULL,
Fecha_de_Matricula DATE NOT NULL,
Nivel_de_Emision VARCHAR(20) NOT NULL,
Pantalla_Info VARCHAR(10) NOT NULL,
Fecha_de_Varada DATE NULL,
PRIMARY KEY (Chasis,Nombre_Concesionario),
CONSTRAINT fk_bt_co FOREIGN KEY(Nombre_Concesionario) REFERENCES Concesionario (Nombre_Concesionario)
ON UPDATE NO ACTION ON DELETE CASCADE
);

/**Se crea la tabla de Padron**/

CREATE TABLE Padron(
Chasis VARCHAR(20) NOT NULL,
Nombre_Concesionario VARCHAR(255) NOT NULL,
Codigo VARCHAR(8) NOT NULL,
Marca VARCHAR(20) NOT NULL,
Placa VARCHAR(6) NOT NULL,
Modelo INT NOT NULL,
Linea VARCHAR(20) NOT NULL,
Accesible VARCHAR(20) NOT NULL,
Fecha_de_Vinculacion DATE NOT NULL,
Fecha_de_Matricula DATE NOT NULL,
Nivel_de_Emision VARCHAR(20) NOT NULL,
CCTV VARCHAR(10) NOT NULL,
Fecha_de_Varada DATE NULL,
PRIMARY KEY (Chasis,Nombre_Concesionario),
CONSTRAINT fk_pd_co FOREIGN KEY(Nombre_Concesionario) REFERENCES Concesionario (Nombre_Concesionario)
ON UPDATE NO ACTION ON DELETE CASCADE
);
